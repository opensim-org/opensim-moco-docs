157  // Set mode.
158  Mode mode;

160  mode = getDefaultMode();
161  } else {

301  MocoAverageSpeedGoal() { constructProperties(); }
302  MocoAverageSpeedGoal(std::string name) : MocoGoal(std::move(name)) {
303  constructProperties();
304  }
305 
306 protected:
307  bool getSupportsEndpointConstraintImpl() const override { return true; }
308  Mode getDefaultModeImpl() const override {
309  return Mode::EndpointConstraint;
310  }
311  void calcGoalImpl(
312  const GoalInput& input, SimTK::Vector& values) const override {
313  SimTK::Real timeInitial = input.initial_state.getTime();
314  SimTK::Real timeFinal = input.final_state.getTime();
315  SimTK::Real duration = timeFinal - timeInitial;
316 
317  SimTK::Vec3 comInitial =
318  getModel().calcMassCenterPosition(input.initial_state);
319  SimTK::Vec3 comFinal =
320  getModel().calcMassCenterPosition(input.final_state);
321  // TODO: Use distance squared for convexity.
322  SimTK::Real displacement = (comFinal - comInitial).norm();
323  // Calculate average gait speed.
324  values[0] = get_desired_average_speed() - (displacement / duration);
325  if (getModeIsCost()) { values[0] = SimTK::square(values[0]); }
326  }
327  void initializeOnModelImpl(const Model&) const override {
328  setNumIntegralsAndOutputs(0, 1);
329  }
330 
331 private:
332  void constructProperties() { constructProperty_desired_average_speed(0); }
333 };
