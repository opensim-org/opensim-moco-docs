310  SimTK::Real calcActiveForceLengthMultiplier(
311  const SimTK::Real& normFiberLength) const {
312  const double& scale = get_active_force_width_scale();
313  // Shift the curve so its peak is at the origin, scale it
314  // horizontally, then shift it back so its peak is still at x = 1.0.
315  const double x = (normFiberLength - 1.0) / scale + 1.0;
316  return calcGaussianLikeCurve(x, b11, b21, b31, b41) +
317  calcGaussianLikeCurve(x, b12, b22, b32, b42) +
318  calcGaussianLikeCurve(x, b13, b23, b33, b43);
319  }

326  SimTK::Real calcActiveForceLengthMultiplierDerivative(
327  const SimTK::Real& normFiberLength) const {
328  const double& scale = get_active_force_width_scale();
329  // Shift the curve so its peak is at the origin, scale it
330  // horizontally, then shift it back so its peak is still at x = 1.0.
331  const double x = (normFiberLength - 1.0) / scale + 1.0;
332  return (1.0 / scale) *
333  (calcGaussianLikeCurveDerivative(x, b11, b21, b31, b41) +
334  calcGaussianLikeCurveDerivative(x, b12, b22, b32, b42) +
335  calcGaussianLikeCurveDerivative(x, b13, b23, b33, b43));
336  }

342  static SimTK::Real calcForceVelocityMultiplier(
343  const SimTK::Real& normFiberVelocity) {
344  using SimTK::square;
345  const SimTK::Real tempV = d2 * normFiberVelocity + d3;
346  const SimTK::Real tempLogArg = tempV + sqrt(square(tempV) + 1.0);
347  return d1 * log(tempLogArg) + d4;
348  }

353  static SimTK::Real calcForceVelocityInverseCurve(
354  const SimTK::Real& forceVelocityMult) {
355  // The version of this equation in the supplementary materials of De
356  // Groote et al., 2016 has an error (it's missing a "-d3" before
357  // dividing by "d2").
358  return (sinh(1.0 / d1 * (forceVelocityMult - d4)) - d3) / d2;

372  SimTK::Real calcPassiveForceMultiplier(
373  const SimTK::Real& normFiberLength) const {
374  if (get_ignore_passive_fiber_force()) return 0;
375 
376  const double& e0 = get_passive_fiber_strain_at_one_norm_force();
377 
378  const double offset =
379  exp(kPE * (m_minNormFiberLength - 1.0) / e0);
380  const double denom = exp(kPE) - offset;
381 
382  return (exp(kPE * (normFiberLength - 1.0) / e0) - offset) / denom;
383  }
384 
387  SimTK::Real calcPassiveForceMultiplierDerivative(
388  const SimTK::Real& normFiberLength) const {
389 
390  if (get_ignore_passive_fiber_force()) return 0;
391 
392  const double& e0 = get_passive_fiber_strain_at_one_norm_force();
393 
394  const double offset = exp(kPE * (m_minNormFiberLength - 1) / e0);
395 
396  return (kPE * exp((kPE * (normFiberLength - 1)) / e0)) /
397  (e0 * (exp(kPE) - offset));
398  }
399 
402  SimTK::Real calcPassiveForceMultiplierIntegral(
403  const SimTK::Real& normFiberLength) const {
404 
405  if (get_ignore_passive_fiber_force()) return 0;
406 
407  const double& e0 = get_passive_fiber_strain_at_one_norm_force();
408 
409  const double temp1 = exp(kPE * m_minNormFiberLength / e0);
410  const double denom = exp(kPE * (1.0 + 1.0 / e0)) - temp1;
411  const double temp2 = kPE / e0 * normFiberLength;
412  return (e0 / kPE * exp(temp2) - normFiberLength * temp1) / denom;
413  }

460  void calcFiberForce(const SimTK::Real& activation,
461  const SimTK::Real& activeForceLengthMultiplier,
462  const SimTK::Real& forceVelocityMultiplier,
463  const SimTK::Real& normPassiveFiberForce,
464  const SimTK::Real& normFiberVelocity, SimTK::Real& activeFiberForce,
465  SimTK::Real& conPassiveFiberForce,
466  SimTK::Real& nonConPassiveFiberForce,
467  SimTK::Real& totalFiberForce) const {
468  const auto& maxIsometricForce = get_max_isometric_force();
469  // active force
470  activeFiberForce =
471  maxIsometricForce * (activation * activeForceLengthMultiplier *
472  forceVelocityMultiplier);
473  // conservative passive force
474  conPassiveFiberForce = maxIsometricForce * normPassiveFiberForce;
475  // non-conservative passive force
476  nonConPassiveFiberForce =
477  maxIsometricForce * get_fiber_damping() * normFiberVelocity;
478  // total force
479  totalFiberForce = activeFiberForce + conPassiveFiberForce +
480  nonConPassiveFiberForce;
481  }

487  SimTK::Real calcFiberStiffness(const SimTK::Real& activation,
488  const SimTK::Real& normFiberLength,
489  const SimTK::Real& fiberVelocityMultiplier) const {
490 
491  const SimTK::Real partialNormFiberLengthPartialFiberLength =
492  1.0 / get_optimal_fiber_length();
493  const SimTK::Real partialNormActiveForcePartialFiberLength =
494  partialNormFiberLengthPartialFiberLength *
495  calcActiveForceLengthMultiplierDerivative(normFiberLength);
496  const SimTK::Real partialNormPassiveForcePartialFiberLength =
497  partialNormFiberLengthPartialFiberLength *
498  calcPassiveForceMultiplierDerivative(normFiberLength);
499 
500  // fiberStiffness = d_fiberForce / d_fiberLength
501  return get_max_isometric_force() *
502  (activation * partialNormActiveForcePartialFiberLength *
503  fiberVelocityMultiplier +
504  partialNormPassiveForcePartialFiberLength);
505  }

532  SimTK::Real calcPartialPennationAnglePartialFiberLength(
533  const SimTK::Real& fiberLength) const {
534 
535  using SimTK::square;
536  // pennationAngle = asin(fiberWidth/fiberLength)
537  // d_pennationAngle/d_fiberLength =
538  // d/d_fiberLength (asin(fiberWidth/fiberLength))
539  return (-m_fiberWidth / square(fiberLength)) /
540  sqrt(1.0 - square(m_fiberWidth / fiberLength));
541  }

547  SimTK::Real calcPartialFiberForceAlongTendonPartialFiberLength(
548  const SimTK::Real& fiberForce, const SimTK::Real& fiberStiffness,
549  const SimTK::Real& sinPennationAngle,
550  const SimTK::Real& cosPennationAngle,
551  const SimTK::Real& partialPennationAnglePartialFiberLength) const {
552 
553  const SimTK::Real partialCosPennationAnglePartialFiberLength =
554  -sinPennationAngle * partialPennationAnglePartialFiberLength;
555 
556  // The stiffness of the fiber along the direction of the tendon. For
557  // small changes in length parallel to the fiber, this quantity is
558  // d_fiberForceAlongTendon / d_fiberLength =
559  // d/d_fiberLength(fiberForce * cosPenneationAngle)
560  return fiberStiffness * cosPennationAngle +
561  fiberForce * partialCosPennationAnglePartialFiberLength;
562  }

568  SimTK::Real calcFiberStiffnessAlongTendon(const SimTK::Real& fiberLength,
569  const SimTK::Real& partialFiberForceAlongTendonPartialFiberLength,
570  const SimTK::Real& sinPennationAngle,
571  const SimTK::Real& cosPennationAngle,
572  const SimTK::Real& partialPennationAnglePartialFiberLength) const {
573 
574  // The change in length of the fiber length along the tendon.
575  // fiberLengthAlongTendon = fiberLength * cosPennationAngle
576  const SimTK::Real partialFiberLengthAlongTendonPartialFiberLength =
577  cosPennationAngle -
578  fiberLength * sinPennationAngle *
579  partialPennationAnglePartialFiberLength;
580 
581  // fiberStiffnessAlongTendon
582  // = d_fiberForceAlongTendon / d_fiberLengthAlongTendon
583  // = (d_fiberForceAlongTendon / d_fiberLength) *
584  // (1 / (d_fiberLengthAlongTendon / d_fiberLength))
585  return partialFiberForceAlongTendonPartialFiberLength *
586  (1.0 / partialFiberLengthAlongTendonPartialFiberLength);
587  }

698  static SimTK::Real calcGaussianLikeCurve(const SimTK::Real& x,
699  const double& b1, const double& b2, const double& b3,
700  const double& b4) {
701  using SimTK::square;
702  return b1 * exp(-0.5 * square(x - b2) / square(b3 + b4 * x));
703  }
704 
707  static SimTK::Real calcGaussianLikeCurveDerivative(const SimTK::Real& x,
708  const double& b1, const double& b2, const double& b3,
709  const double& b4) {
710  using SimTK::cube;
711  using SimTK::square;
712  return (b1 * exp(-square(b2 - x) / (2 * square(b3 + b4 * x))) *
713  (b2 - x) * (b3 + b2 * b4)) /
714  cube(b3 + b4 * x);
715  }
