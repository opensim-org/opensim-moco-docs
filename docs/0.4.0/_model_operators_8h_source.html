71 class OSIMMOCO_API ModOpTendonComplianceDynamicsModeDGF
72  : public ModelOperator {
73  OpenSim_DECLARE_CONCRETE_OBJECT(
74  ModOpTendonComplianceDynamicsModeDGF, ModelOperator);
75  OpenSim_DECLARE_PROPERTY(mode, std::string,
76  "The tendon compliance dynamics mode: 'implicit' or 'explicit'. "
77  "Default: 'explicit'.");
78 
79 public:
80  ModOpTendonComplianceDynamicsModeDGF() {
81  constructProperty_mode("explicit");
82  }
83  ModOpTendonComplianceDynamicsModeDGF(std::string mode) :
84  ModOpTendonComplianceDynamicsModeDGF() {
85  OPENSIM_THROW_IF(mode != "explicit" && mode != "implicit", Exception,
86  format("The tendon compliance dynamics mode must be either "
87  "'explicit' or 'implicit', but %s was provided.", mode));
88  set_mode(std::move(mode));
89  }
90  void operate(Model& model, const std::string&) const override {
91  model.finalizeFromProperties();
92  for (auto& muscle :
93  model.updComponentList<DeGrooteFregly2016Muscle>()) {
94  if (!muscle.get_ignore_tendon_compliance()) {
95  muscle.set_tendon_compliance_dynamics_mode(get_mode());
96  }
97  }
98  }

297  }
299  void operate(Model& model,
300  const std::string& relativeToDirectory) const override {
301  std::string path = get_filepath();
302  if (!relativeToDirectory.empty()) {
