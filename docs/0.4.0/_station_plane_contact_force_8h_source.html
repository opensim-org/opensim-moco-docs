98  SimTK::Vec3 calcContactForceOnStation(const SimTK::State& s)
99  const override {
100  SimTK::Vec3 force(0);
101  const auto& pt = getConnectee<Station>("station");
102  const auto& pos = pt.getLocationInGround(s);
103  const auto& vel = pt.getVelocityInGround(s);
104  const SimTK::Real y = pos[1];
105  const SimTK::Real velNormal = vel[1];
106  // TODO should project vel into ground.
107  const SimTK::Real velSliding = vel[0];
108  const SimTK::Real depth = 0 - y;
109  const SimTK::Real depthRate = 0 - velNormal;
110  const SimTK::Real& a = get_stiffness();
111  const SimTK::Real& b = get_dissipation();
112  if (depth > 0) {
113  force[1] = fmax(0, a * pow(depth, 3) * (1 + b * depthRate));
114  }
115  const SimTK::Real voidStiffness = 1.0; // N/m
116  force[1] += voidStiffness * depth;
117 
118  const SimTK::Real velSlidingScaling =
119  get_tangent_velocity_scaling_factor();
120  // The paper used (1 - exp(-x)) / (1 + exp(-x)) = tanh(2x).
121  // tanh() has a wider domain than using exp().
122  const SimTK::Real transition = tanh(velSliding / velSlidingScaling / 2);
123 
124  const SimTK::Real frictionForce =
125  -transition * get_friction_coefficient() * force[1];
126 
127  force[0] = frictionForce;
128  return force;
129  }
130 
131 private:
132  void constructProperties() {
133  constructProperty_friction_coefficient(1.0);
134  constructProperty_stiffness(5e7);
135  constructProperty_dissipation(1.0);
136  constructProperty_tangent_velocity_scaling_factor(0.05);
137  }

165  SimTK::Vec3 calcContactForceOnStation(const SimTK::State& s)
166  const override {
167  SimTK::Vec3 force(0);
168  const auto& pt = getConnectee<Station>("station");
169  const auto& pos = pt.getLocationInGround(s);
170  const auto& vel = pt.getVelocityInGround(s);
171  const SimTK::Real y = pos[1];
172  const SimTK::Real velNormal = vel[1];
173  // TODO should project vel into ground.
174  const SimTK::Real velSliding = vel[0];
175  // const SimTK::Real depth = 0 - y;
176  const SimTK::Real depthRate = 0 - velNormal;
177  const SimTK::Real Kval = get_stiffness();
178  const SimTK::Real Cval = get_dissipation();
179  const SimTK::Real tscale = get_tscale();
180  const SimTK::Real klow = 1e-1 / (tscale * tscale);
181  const SimTK::Real h = 1e-3;
182  const SimTK::Real c = 5e-4;
183  const SimTK::Real ymax = 1e-2;
184 
186  const SimTK::Real vp = (Kval + klow) / (Kval - klow);
187  const SimTK::Real sp = (Kval - klow) / 2;
188 
189  const SimTK::Real constant =
190  -sp * (vp * ymax - c * log(cosh((ymax + h) / c)));
191 
192  SimTK::Real Fspring =
193  -sp * (vp * y - c * log(cosh((y + h) / c))) - constant;
194  if (SimTK::isNaN(Fspring) || SimTK::isInf(Fspring)) {
195  Fspring = 0;
196  }
197 
198  const SimTK::Real Fy = Fspring * (1 + Cval * depthRate);
199 
200  force[1] = Fy;
201 
203  const SimTK::Real mu_d = 1;
204  const SimTK::Real latchvel = 0.05; // m/s
205 
206  const SimTK::Real mu = mu_d * tanh(velSliding / latchvel / 2);
207  force[0] = -force[1] * mu;
208 
209  return force;
210  }
211 
212 private:
213  void constructProperties() {
214  constructProperty_stiffness(1e4);
215  constructProperty_dissipation(1e-2);
216  constructProperty_tscale(1.0);
217  }

262  SimTK::Vec3 calcContactForceOnStation(const SimTK::State& s)
263  const override {
264  using SimTK::square;
265  SimTK::Vec3 force(0);
266  const auto& pt = getConnectee<Station>("station");
267  const auto& pos = pt.getLocationInGround(s);
268  const auto& vel = pt.getVelocityInGround(s);
269  const SimTK::Real height = pos[1];
270  const SimTK::Real velNormal = vel[1];
271  // TODO should project vel into ground.
272  const SimTK::Real velSliding = vel[0];
273 
274  // The Appendix of Esposito and Miller 2018 uses height above ground,
275  // but we use penetration depth, so some signs are reversed.
276 
277  // Normal force.
278  const SimTK::Real depth = 0 - height;
279  const SimTK::Real depthRate = 0 - velNormal;
280  const SimTK::Real a = get_stiffness();
281  const SimTK::Real b = get_dissipation();
282 
283  // dy approaches 0 as y -> inf
284  // depth as y -> -inf
285  const SimTK::Real dy =
286  0.5 * (sqrt(square(depth) + m_depthOffsetSquared) + depth);
287  const SimTK::Real voidStiffness = 1.0; // N/m
288  force[1] = a * square(dy) * (1 + b * depthRate) + voidStiffness * depth;
289 
290  // Friction (TODO handle 3D).
291  const SimTK::Real velSlidingScaling =
292  get_tangent_velocity_scaling_factor();
293  const SimTK::Real transition = tanh(velSliding / velSlidingScaling);
294 
295  const SimTK::Real frictionForce =
296  -transition * get_friction_coefficient() * force[1];
297  force[0] = frictionForce;
298  return force;
299  }
300 
301  // TODO potential energy.
302 private:
303  void constructProperties() {
304  constructProperty_stiffness(2.0e6);
305  constructProperty_dissipation(1.0);
306  constructProperty_friction_coefficient(1.0);
307  constructProperty_tangent_velocity_scaling_factor(0.05);
308  constructProperty_depth_offset(0.001);
309  }
310  SimTK::Real m_depthOffsetSquared;
311 };
