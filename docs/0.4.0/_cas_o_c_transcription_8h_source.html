203  CasOC::VariablesDM expandVariables(const casadi::DM& x) const {
204  CasOC::VariablesDM out;
205  using casadi::Slice;
206  casadi_int offset = 0;
207  for (const auto& key : getSortedVarKeys(m_vars)) {
208  const auto& value = m_vars.at(key);
209  // Convert a portion of the column vector into a matrix.
210  out[key] = casadi::DM::reshape(
211  x(Slice(offset, offset + value.numel())), value.rows(),
212  value.columns());
213  offset += value.numel();
214  }
215  return out;
216  }

221  template <typename T>
222  T flattenConstraints(const Constraints<T>& constraints) const {
223  T flat = T(casadi::Sparsity::dense(m_numConstraints, 1));
224 
225  int iflat = 0;
226  auto copyColumn = [&flat, &iflat](const T& matrix, int columnIndex) {
227  using casadi::Slice;
228  if (matrix.rows()) {
229  flat(Slice(iflat, iflat + matrix.rows())) =
230  matrix(Slice(), columnIndex);
231  iflat += matrix.rows();
232  }
233  };

235  // Trapezoidal sparsity pattern for mesh intervals 0, 1 and 2:
236  // Endpoint constraints depend on all time points through their
237  // integral.
238  // 0 1 2 3
239  // endpoint x x x x
240  // path_0 x
241  // kinematic_0 x
242  // residual_0 x
243  // defect_0 x x
244  // kinematic_1 x
245  // path_1 x
246  // residual_1 x
247  // defect_1 x x
248  // kinematic_2 x
249  // path_2 x
250  // residual_2 x
251  // kinematic_3 x
252  // path_3 x
253  // residual_3 x
254 
255  // Hermite-Simpson sparsity pattern for mesh intervals 0, 1 and 2:
256  // 0 0.5 1 1.5 2 2.5 3
257  // endpoint x x x x x x x
258  // path_0 x
259  // kinematic_0 x
260  // residual_0 x
261  // residual_0.5 x
262  // defect_0 x x x
263  // interp_con_0 x x x
264  // kinematic_1 x
265  // path_1 x
266  // residual_1 x
267  // residual_1.5 x
268  // defect_1 x x x
269  // interp_con_1 x x x
270  // kinematic_2 x
271  // path_2 x
272  // residual_2 x
273  // residual_2.5 x
274  // defect_2 x x x
275  // interp_con_2 x x x
276  // kinematic_3 x
277  // path_3 x
278  // residual_3 x
279  // 0 0.5 1 1.5 2 2.5 3
280 
281  for (const auto& endpoint : constraints.endpoint) {
282  copyColumn(endpoint, 0);
283  }
284 
285  int igrid = 0;
286  // Index for pointsForInterpControls.
287  int icon = 0;
288  for (int imesh = 0; imesh < m_numMeshPoints; ++imesh) {
289  copyColumn(constraints.kinematic, imesh);
290  for (const auto& path : constraints.path) {
291  copyColumn(path, imesh);
292  }
293  if (imesh < m_numMeshIntervals) {
294  while (m_grid(igrid).scalar() < m_solver.getMesh()[imesh + 1]) {
295  copyColumn(constraints.multibody_residuals, igrid);
296  copyColumn(constraints.auxiliary_residuals, igrid);
297  ++igrid;
298  }
299  copyColumn(constraints.defects, imesh);
300  while (icon < m_pointsForInterpControls.numel() &&
301  m_pointsForInterpControls(icon).scalar() <
302  m_solver.getMesh()[imesh + 1]) {
303  copyColumn(constraints.interp_controls, icon);
304  ++icon;
305  }
306  }
307  }
308  // The loop above does not handle the residual at the final grid point.
309  copyColumn(constraints.multibody_residuals, m_numGridPoints - 1);
310  copyColumn(constraints.auxiliary_residuals, m_numGridPoints - 1);
311 
312  OPENSIM_THROW_IF(iflat != m_numConstraints, OpenSim::Exception,
313  "Internal error.");
314  return flat;
315  }

317  // Expand constraints that have been flattened into a Constraints struct.
318  template <typename T>
319  Constraints<T> expandConstraints(const T& flat) const {
320  using casadi::Sparsity;
321 
322  // Allocate memory.
323  auto init = [](int numRows, int numColumns) {
324  return T(casadi::Sparsity::dense(numRows, numColumns));
325  };
326  Constraints<T> out;
327  out.defects = init(m_numDefectsPerMeshInterval, m_numMeshPoints - 1);
328  out.multibody_residuals = init(m_numMultibodyResiduals,
329  m_numGridPoints);
330  out.auxiliary_residuals = init(m_numAuxiliaryResiduals,
331  m_numGridPoints);
332  out.kinematic = init(m_problem.getNumKinematicConstraintEquations(),
333  m_numMeshPoints);
334  out.endpoint.resize(m_problem.getEndpointConstraintInfos().size());
335  for (int iec = 0; iec < (int)m_constraints.endpoint.size(); ++iec) {
336  const auto& info = m_problem.getEndpointConstraintInfos()[iec];
337  out.endpoint[iec] = init(info.num_outputs, 1);
338  }
339  out.path.resize(m_problem.getPathConstraintInfos().size());
340  for (int ipc = 0; ipc < (int)m_constraints.path.size(); ++ipc) {
341  const auto& info = m_problem.getPathConstraintInfos()[ipc];
342  out.path[ipc] = init(info.size(), m_numMeshPoints);
343  }
344  out.interp_controls = init(m_problem.getNumControls(),
345  (int)m_pointsForInterpControls.numel());
346 
347  int iflat = 0;
348  auto copyColumn = [&flat, &iflat](T& matrix, int columnIndex) {
349  using casadi::Slice;
350  if (matrix.rows()) {
351  matrix(Slice(), columnIndex) =
352  flat(Slice(iflat, iflat + matrix.rows()));
353  iflat += matrix.rows();
354  }
355  };
356 
357  for (auto& endpoint : out.endpoint) {
358  copyColumn(endpoint, 0);
359  }
360 
361  int igrid = 0;
362  // Index for pointsForInterpControls.
363  int icon = 0;
364  for (int imesh = 0; imesh < m_numMeshPoints; ++imesh) {
365  copyColumn(out.kinematic, imesh);
366  for (auto& path : out.path) { copyColumn(path, imesh); }
367  if (imesh < m_numMeshIntervals) {
368  while (m_grid(igrid).scalar() < m_solver.getMesh()[imesh + 1]) {
369  copyColumn(out.multibody_residuals, igrid);
370  copyColumn(out.auxiliary_residuals, igrid);
371  ++igrid;
372  }
373  copyColumn(out.defects, imesh);
374  while (icon < m_pointsForInterpControls.numel() &&
375  m_pointsForInterpControls(icon).scalar() <
376  m_solver.getMesh()[imesh + 1]) {
377  copyColumn(out.interp_controls, icon);
378  ++icon;
379  }
380  }
381  }
382  // The loop above does not handle residuals at the final grid point.
383  copyColumn(out.multibody_residuals, m_numGridPoints - 1);
384  copyColumn(out.auxiliary_residuals, m_numGridPoints - 1);
385 
386  OPENSIM_THROW_IF(iflat != m_numConstraints, OpenSim::Exception,
387  "Internal error.");
388  return out;
389  }
