31 class Function : public casadi::Callback {
32 public:
33  virtual ~Function() = default;
34  void constructFunction(const Problem* casProblem, const std::string& name,
35  const std::string& finiteDiffScheme,
36  std::shared_ptr<const std::vector<VariablesDM>>
37  pointsForSparsityDetection);
38  void setCommonOptions(casadi::Dict& opts) {
39  // Compute the derivatives of this function using finite differences.
40  opts["enable_fd"] = true;
41  opts["fd_method"] = getFiniteDifferenceScheme();
42  // Using "forward", iterations are 10x faster but problems are less
43  // likely to converge.
44  }
45  std::string getFiniteDifferenceScheme() {
46  return m_finite_difference_scheme;
47  }
48  casadi_int get_n_in() override { return 6; }
49  std::string get_name_in(casadi_int i) override {
50  switch (i) {
51  case 0: return "time";
52  case 1: return "states";
53  case 2: return "controls";
54  case 3: return "multipliers";
55  case 4: return "derivatives";
56  case 5: return "parameters";
57  default: OPENSIM_THROW(OpenSim::Exception, "Internal error.");
58  }
59  }
60  casadi::Sparsity get_sparsity_in(casadi_int i) override;
61  bool has_jacobian_sparsity() const override {
62  return !m_fullPointsForSparsityDetection->empty();
63  }
64  casadi::Sparsity get_jacobian_sparsity() const override;
65 
66 protected:
67  const Problem* m_casProblem;
68 
69 private:
72  VectorDM getSubsetPointsForSparsityDetection() const {
73  VectorDM out(m_fullPointsForSparsityDetection->size());
74  for (int i = 0; i < (int)out.size(); ++i) {
75  out[i] = getSubsetPoint(m_fullPointsForSparsityDetection->at(i));
76  }
77  return out;
78  }
79  virtual casadi::DM getSubsetPoint(const VariablesDM& fullPoint) const {
80  int itime = 0;
81  using casadi::Slice;
82  return casadi::DM::vertcat({fullPoint.at(initial_time),
83  fullPoint.at(states)(Slice(), itime),
84  fullPoint.at(controls)(Slice(), itime),
85  fullPoint.at(multipliers)(Slice(), itime),
86  fullPoint.at(derivatives)(Slice(), itime),
87  fullPoint.at(parameters)});
88  }
89 
90  std::string m_finite_difference_scheme = "central";
91 
92  std::shared_ptr<const std::vector<VariablesDM>>
93  m_fullPointsForSparsityDetection;
94 };

211  }

220  using casadi::Slice;
221  return casadi::DM::vertcat({fullPoint.at(initial_time),
222  fullPoint.at(states)(Slice(), 0),
223  fullPoint.at(controls)(Slice(), 0),
224  fullPoint.at(multipliers)(Slice(), 0),
225  fullPoint.at(derivatives)(Slice(), 0), fullPoint.at(final_time),
226  fullPoint.at(states)(Slice(), -1),
227  fullPoint.at(controls)(Slice(), -1),
228  fullPoint.at(multipliers)(Slice(), -1),
229  fullPoint.at(derivatives)(Slice(), -1),
230  fullPoint.at(parameters),
231  // TODO: We should find a way to actually compute the integral

235  }
236 protected:
237  int m_index = -1;
238  int m_numEquations = -1;
