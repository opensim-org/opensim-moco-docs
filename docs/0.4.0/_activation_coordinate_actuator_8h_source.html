76  // TODO no need to do clamping, etc; CoordinateActuator is bidirectional.
77  void computeStateVariableDerivatives(const SimTK::State& s) const override
78  {
79  const auto& tau = get_activation_time_constant();
80  const auto& u = getControl(s);
81  const auto& a = getStateVariableValue(s, "activation");
82  const SimTK::Real adot = (u - a) / tau;
83  setStateVariableDerivativeValue(s, "activation", adot);
84  }
