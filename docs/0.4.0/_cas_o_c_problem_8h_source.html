260  void addEndpointConstraint(
261  std::string name, int numIntegrals, std::vector<Bounds> bounds) {
262  OPENSIM_THROW_IF(numIntegrals < 0 || numIntegrals > 1,
263  OpenSim::Exception, "numIntegrals must be 0 or 1.");
264  std::unique_ptr<EndpointConstraintIntegrand> integrand_function;
265  if (numIntegrals) {
266  integrand_function =
267  OpenSim::make_unique<EndpointConstraintIntegrand>();
268  }
269  casadi::DM lower(bounds.size(), 1);
270  casadi::DM upper(bounds.size(), 1);
271  for (int ibound = 0; ibound < (int)bounds.size(); ++ibound) {
272  lower(ibound, 0) = bounds[ibound].lower;
273  upper(ibound, 0) = bounds[ibound].upper;
274  }
275  m_endpointConstraintInfos.emplace_back(std::move(name),
276  (int)bounds.size(), std::move(integrand_function),
277  OpenSim::make_unique<EndpointConstraint>(), std::move(lower),
278  std::move(upper));
279  }

384  void initialize(const std::string& finiteDiffScheme,
385  std::shared_ptr<const std::vector<VariablesDM>>
386  pointsForSparsityDetection) const {
387  auto* mutThis = const_cast<Problem*>(this);
388 
389  {
390  int index = 0;
391  for (const auto& costInfo : mutThis->m_costInfos) {
392  costInfo.endpoint_function->constructFunction(this,
393  "cost_" + costInfo.name + "_endpoint", index,
394  costInfo.num_outputs, finiteDiffScheme,
395  pointsForSparsityDetection);
396  if (costInfo.integrand_function) {
397  costInfo.integrand_function->constructFunction(this,
398  "cost_" + costInfo.name + "_integrand", index,
399  finiteDiffScheme, pointsForSparsityDetection);
400  }
401  ++index;
402  }
403  }
404  {
405  int index = 0;
406  for (const auto& info : mutThis->m_endpointConstraintInfos) {
407  info.endpoint_function->constructFunction(this,
408  "endpoint_constraint_" + info.name + "_endpoint", index,
409  info.num_outputs, finiteDiffScheme,
410  pointsForSparsityDetection);
411  if (info.integrand_function) {
412  info.integrand_function->constructFunction(this,
413  "endpoint_constraint_" + info.name + "_integrand", index,
414  finiteDiffScheme, pointsForSparsityDetection);
415  }
416  ++index;
417  }
418  }
419  {
420  int index = 0;
421  for (const auto& pathInfo : mutThis->m_pathInfos) {
422  pathInfo.function->constructFunction(this,
423  "path_constraint_" + pathInfo.name, index,
424  (int)pathInfo.lowerBounds.size1(), finiteDiffScheme,
425  pointsForSparsityDetection);
426  ++index;
427  }
428  }
429 
430  if (m_dynamicsMode == "implicit") {
431  // Construct a full implicit multibody system (i.e. including
432  // kinematic constraints).
433  mutThis->m_implicitMultibodyFunc =
434  OpenSim::make_unique<MultibodySystemImplicit<true>>();
435  mutThis->m_implicitMultibodyFunc->constructFunction(this,
436  "implicit_multibody_system", finiteDiffScheme,
437  pointsForSparsityDetection);
438 
439  // Construct an implicit multibody system ignoring kinematic
440  // constraints.
441  mutThis->m_implicitMultibodyFuncIgnoringConstraints =
442  OpenSim::make_unique<MultibodySystemImplicit<false>>();
443  mutThis->m_implicitMultibodyFuncIgnoringConstraints
444  ->constructFunction(this,
445  "implicit_multibody_system_ignoring_constraints",
446  finiteDiffScheme, pointsForSparsityDetection);
447  } else {
448  mutThis->m_multibodyFunc =
449  OpenSim::make_unique<MultibodySystemExplicit<true>>();
450  mutThis->m_multibodyFunc->constructFunction(this,
451  "explicit_multibody_system", finiteDiffScheme,
452  pointsForSparsityDetection);
453 
454  mutThis->m_multibodyFuncIgnoringConstraints =
455  OpenSim::make_unique<MultibodySystemExplicit<false>>();
456  mutThis->m_multibodyFuncIgnoringConstraints->constructFunction(this,
457  "multibody_system_ignoring_constraints", finiteDiffScheme,
458  pointsForSparsityDetection);
459  }
460 
461  if (m_enforceConstraintDerivatives) {
462  mutThis->m_velocityCorrectionFunc =
463  OpenSim::make_unique<VelocityCorrection>();
464  mutThis->m_velocityCorrectionFunc->constructFunction(this,
465  "velocity_correction", finiteDiffScheme,
466  pointsForSparsityDetection);
467  }
468  }
