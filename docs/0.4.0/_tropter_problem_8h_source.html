128  void addKinematicConstraints() {
129  // Add any scalar constraints associated with kinematic constraints in
130  // the model as path constraints in the problem.
131  // Whether or not enabled kinematic constraints exist in the model,
132  // check that optional solver properties related to constraints are
133  // set properly.
134  const std::vector<std::string> kcNames =
135  m_mocoProbRep.createKinematicConstraintNames();
136  if (kcNames.empty()) {
137  OPENSIM_THROW_IF(
138  m_mocoTropterSolver.get_minimize_lagrange_multipliers(),
139  Exception,
140  "Solver property 'minimize_lagrange_multipliers' "
141  "was enabled but no enabled kinematic constraints exist in "
142  "the "
143  "model.");
144  // Do not add kinematic constraints, so we can return.
145  return;
146  }
147 
148  int cid, mp, mv, ma;
149  int numEquationsThisConstraint;
150  int multIndexThisConstraint;
151  std::vector<MocoBounds> bounds;
152  std::vector<std::string> labels;
153  std::vector<KinematicLevel> kinLevels;
154  const bool enforceConstraintDerivs =
155  m_mocoTropterSolver.get_enforce_constraint_derivatives();
156  for (const auto& kcName : kcNames) {
157  const auto& kc = m_mocoProbRep.getKinematicConstraint(kcName);
158  const auto& multInfos = m_mocoProbRep.getMultiplierInfos(kcName);
159  cid = kc.getSimbodyConstraintIndex();
160  mp = kc.getNumPositionEquations();
161  mv = kc.getNumVelocityEquations();
162  ma = kc.getNumAccelerationEquations();
163  bounds = kc.getConstraintInfo().getBounds();
164  labels = kc.getConstraintInfo().getConstraintLabels();
165  kinLevels = kc.getKinematicLevels();
166 
167  // TODO only add velocity correction variables for holonomic
168  // constraint derivatives? For now, disallow enforcing derivatives
169  // if non-holonomic or acceleration constraints present.
170  OPENSIM_THROW_IF(enforceConstraintDerivs && mv != 0, Exception,
171  format("Enforcing constraint derivatives is supported only "
172  "for "
173  "holonomic (position-level) constraints. "
174  "There are %i velocity-level "
175  "scalar constraints associated with the model "
176  "Constraint "
177  "at ConstraintIndex %i.",
178  mv, cid));
179  OPENSIM_THROW_IF(enforceConstraintDerivs && ma != 0, Exception,
180  format("Enforcing constraint derivatives is supported only "
181  "for "
182  "holonomic (position-level) constraints. "
183  "There are %i acceleration-level "
184  "scalar constraints associated with the model "
185  "Constraint "
186  "at ConstraintIndex %i.",
187  ma, cid));
188 
189  m_total_mp += mp;
190  m_total_mv += mv;
191  m_total_ma += ma;
192 
193  // Loop through all scalar constraints associated with the model
194  // constraint and corresponding path constraints to the optimal
195  // control problem.
196  //
197  // We need a different index for the Lagrange multipliers since
198  // they are only added if the current constraint equation is not a
199  // derivative of a position- or velocity-level equation.
200  multIndexThisConstraint = 0;
201  numEquationsThisConstraint = 0;
202  for (int i = 0; i < kc.getConstraintInfo().getNumEquations(); ++i) {
203 
204  // If the index for this path constraint represents an
205  // a non-derivative scalar constraint equation, also add a
206  // Lagrange multiplier to the problem.
207  if (kinLevels[i] == KinematicLevel::Position ||
208  kinLevels[i] == KinematicLevel::Velocity ||
209  kinLevels[i] == KinematicLevel::Acceleration) {
210 
211  // TODO name constraints based on model constraint names
212  // or coordinate names if a locked or prescribed coordinate
213  this->add_path_constraint(
214  labels[i], convertBounds(bounds[i]));
215 
216  const auto& multInfo = multInfos[multIndexThisConstraint];
217  this->add_adjunct(multInfo.getName(),
218  convertBounds(multInfo.getBounds()),
219  convertBounds(multInfo.getInitialBounds()),
220  convertBounds(multInfo.getFinalBounds()));
221  // Add velocity correction variables if enforcing
222  // constraint equation derivatives.
223  if (enforceConstraintDerivs) {
224  // TODO this naming convention assumes that the
225  // associated Lagrange multiplier name begins with
226  // "lambda", which may change in the future.
227  OPENSIM_THROW_IF(
228  multInfo.getName().substr(0, 6) != "lambda",
229  Exception,
230  format("Expected the multiplier name for this "
231  "constraint to begin with 'lambda' but "
232  "it "
233  "begins with '%s'.",
234  multInfo.getName().substr(0, 6)));
235  this->add_diffuse(std::string(multInfo.getName())
236  .replace(0, 6, "gamma"),
237  convertBounds(
238  m_mocoTropterSolver
239  .get_velocity_correction_bounds()));
240  }
241  ++multIndexThisConstraint;
242  ++numEquationsThisConstraint;
243 
244  // If enforcing constraint derivatives, also add path
245  // constraints for kinematic constraint equation
246  // derivatives.
247  } else if (enforceConstraintDerivs) {
248  this->add_path_constraint(
249  labels[i], convertBounds(bounds[i]));
250  ++numEquationsThisConstraint;
251  }
252  }
253 
254  m_numKinematicConstraintEquations += numEquationsThisConstraint;
255  }
256  m_numMultipliers = m_total_mp + m_total_mv + m_total_ma;
257  }

315  void setSimTKState(const T& time,
316  const Eigen::Ref<const tropter::VectorX<T>>& states,
317  const Eigen::Ref<const tropter::VectorX<T>>& controls,
318  const Eigen::Ref<const tropter::VectorX<T>>& adjuncts,
319  int stateDisConIndex = 0) const {
320 
321  auto& simTKStateBase = this->m_stateBase;
322  auto& simTKStateDisabledConstraints =
323  m_mocoProbRep.updStateDisabledConstraints(stateDisConIndex);
324  auto& modelDisabledConstraints = this->m_modelDisabledConstraints;
325 
326  if (m_implicit && !m_mocoProbRep.isPrescribedKinematics()) {
327  const auto& accel = this->m_mocoProbRep.getAccelerationMotion();
328  const int NU = simTKStateDisabledConstraints.getNU();
329  const auto& w = adjuncts.segment(
330  this->m_numKinematicConstraintEquations, NU);
331  SimTK::Vector udot((int)w.size(), w.data(), true);
332  accel.setUDot(simTKStateDisabledConstraints, udot);
333  }
334 
335  this->setSimTKTimeAndStates(
336  time, states, simTKStateDisabledConstraints);
337 
338  if (modelDisabledConstraints.getNumControls()) {
339  // Set the controls for actuators in the OpenSim model with disabled
340  // constraints. The base model never gets realized past
341  // Stage::Velocity, so we don't ever need to set its controls.
342  auto& osimControls = modelDisabledConstraints.updControls(
343  simTKStateDisabledConstraints);
344  for (int ic = 0; ic < controls.size(); ++ic) {
345  osimControls[m_modelControlIndices[ic]] = controls[ic];
346  }
347  modelDisabledConstraints.realizeVelocity(
348  simTKStateDisabledConstraints);
349  modelDisabledConstraints.setControls(
350  simTKStateDisabledConstraints, osimControls);
351  }
352 
353  // If enabled constraints exist in the model, compute constraint forces
354  // based on Lagrange multipliers, update the associated
355  // discrete variables in the state.
356  if (this->m_numKinematicConstraintEquations) {
357  this->setSimTKTimeAndStates(time, states, simTKStateBase);
358  this->calcAndApplyKinematicConstraintForces(
359  adjuncts, simTKStateBase, simTKStateDisabledConstraints);
360  }
361  }

363  void calc_cost_integrand(int cost_index, const tropter::Input<T>& in,
364  T& integrand) const override {
365  if (cost_index == m_multiplierCostIndex) {
366  // Unpack variables.
367  const auto& adjuncts = in.adjuncts;
368  // If specified, add squared multiplers cost to the integrand.
369  const auto& multiplierWeight =
370  m_mocoTropterSolver.get_lagrange_multiplier_weight();
371  for (int i = 0; i < m_numMultipliers; ++i) {
372  integrand += multiplierWeight * adjuncts[i] * adjuncts[i];
373  }
374  return;
375  }
376 
377  // Update the state.
378  // TODO would it make sense to a vector of States, one for each mesh
379  // point, so that each can preserve their cache?
380  this->setSimTKState(in);
381 
382  // Compute the integrand for this cost term.
383  const auto& cost = m_mocoProbRep.getCostByIndex(cost_index);
384  integrand = cost.calcIntegrand(this->m_stateDisabledConstraints);
385  }

387  void calc_cost(int cost_index, const tropter::CostInput<T>& in,
388  T& cost_value) const override {
389  if (cost_index == m_multiplierCostIndex) {
390  cost_value = in.integral;
391  return;
392  }
393 
394  // Update the state.
395  this->setSimTKStateForCostInitial(in);
396  this->setSimTKStateForCostFinal(in);
397 
398  const auto& initialState = m_mocoProbRep.updStateDisabledConstraints(0);
399  const auto& finalState = m_mocoProbRep.updStateDisabledConstraints(1);
400 
401  // Compute the cost for this cost term.
402  const auto& cost = m_mocoProbRep.getCostByIndex(cost_index);
403  SimTK::Vector costVector(cost.getNumOutputs());
404  cost.calcGoal({initialState, finalState, in.integral}, costVector);
405  cost_value = costVector.sum();
406  }

461  void calcAndApplyKinematicConstraintForces(
462  const tropter::VectorX<T>& adjuncts, const SimTK::State& stateBase,
463  SimTK::State& stateDisabledConstraints) const {
464  // Calculate the constraint forces using the original model and the
465  // solver-provided Lagrange multipliers.
466  m_modelBase.realizeVelocity(stateBase);
467  const auto& matter = m_modelBase.getMatterSubsystem();
468  // Multipliers are negated so constraint forces can be used like
469  // applied forces.
470  SimTK::Vector multipliers(m_numMultipliers, adjuncts.data(), true);
471  matter.calcConstraintForcesFromMultipliers(stateBase, -multipliers,
472  m_constraintBodyForces, m_constraintMobilityForces);
473  // Apply the constraint forces on the model with disabled constraints.
474  const auto& constraintForces = m_mocoProbRep.getConstraintForces();
475  constraintForces.setAllForces(stateDisabledConstraints,
476  m_constraintMobilityForces, m_constraintBodyForces);
477  }

479  void calcKinematicConstraintErrors(
480  const SimTK::Vector& udot, tropter::Output<T>& out) const {
481  // Only compute constraint errors if we're at a time point where path
482  // constraints in the optimal control problem are enforced.
483  if (out.path.size() != 0 && this->m_numKinematicConstraintEquations) {
484  auto& stateBase = this->m_stateBase;
485 
486  // Position-level errors.
487  std::copy_n(stateBase.getQErr().getContiguousScalarData(),
488  m_total_mp, out.path.data());
489 
490  const auto& enforceConstraintDerivatives =
491  m_mocoTropterSolver.get_enforce_constraint_derivatives();
492  if (enforceConstraintDerivatives || m_total_ma) {
493  // Calculuate udoterr. We cannot use State::getUDotErr()
494  // because that uses Simbody's multiplilers and UDot,
495  // whereas we have our own multipliers and UDot. Here, we use
496  // the udot computed from the model with disabled constraints
497  // since we cannot use (nor do we have availabe) udot computed
498  // from the original model.
499  const auto& matterBase = m_modelBase.getMatterSubsystem();
500  matterBase.calcConstraintAccelerationErrors(
501  stateBase, udot, m_pvaerr);
502  } else {
503  m_pvaerr = SimTK::NaN;
504  }
505 
506  if (enforceConstraintDerivatives) {
507  // Velocity-level errors.
508  std::copy_n(stateBase.getUErr().getContiguousScalarData(),
509  m_total_mp + m_total_mv, out.path.data() + m_total_mp);
510  // Acceleration-level errors.
511  std::copy_n(m_pvaerr.getContiguousScalarData(),
512  m_total_mp + m_total_mv + m_total_ma,
513  out.path.data() + 2 * m_total_mp + m_total_mv);
514  } else {
515  // Velocity-level errors. Skip derivatives of position-level
516  // constraint equations.
517  std::copy_n(stateBase.getUErr().getContiguousScalarData() +
518  m_total_mp,
519  m_total_mv, out.path.data() + m_total_mp);
520  // Acceleration-level errors. Skip derivatives of velocity-
521  // and position-level constraint equations.
522  std::copy_n(m_pvaerr.getContiguousScalarData() + m_total_mp +
523  m_total_mv,
524  m_total_ma, out.path.data() + m_total_mp + m_total_mv);
525  }
526  }
527  }

554 template <typename T>
555 class MocoTropterSolver::ExplicitTropterProblem
556  : public MocoTropterSolver::TropterProblemBase<T> {
557 public:
558  ExplicitTropterProblem(const MocoTropterSolver& solver)
559  : MocoTropterSolver::TropterProblemBase<T>(solver) {}
560  void initialize_on_mesh(const Eigen::VectorXd&) const override {}
561  void calc_differential_algebraic_equations(const tropter::Input<T>& in,
562  tropter::Output<T> out) const override {
563  // Unpack variables.
564  const auto& diffuses = in.diffuses;
565 
566  // Original model and its associated state. These are used to calculate
567  // kinematic constraint forces and errors.
568  const auto& modelBase = this->m_modelBase;
569  auto& simTKStateBase = this->m_stateBase;
570 
571  // Model with disabled constraints and its associated state. These are
572  // used to compute the accelerations.
573  const auto& modelDisabledConstraints = this->m_modelDisabledConstraints;
574  auto& simTKStateDisabledConstraints = this->m_stateDisabledConstraints;
575 
576  // Update the state.
577  this->setSimTKState(in);
578 
579  // Compute the accelerations.
580  // TODO Antoine and Gil said realizing Dynamics is a lot costlier
581  // than realizing to Velocity and computing forces manually.
582  modelDisabledConstraints.realizeAcceleration(
583  simTKStateDisabledConstraints);
584 
585  // Compute kinematic constraint errors if they exist.
586  this->calcKinematicConstraintErrors(
587  simTKStateDisabledConstraints.getUDot(), out);
588 
589  // Apply velocity correction to qdot if at a mesh interval midpoint.
590  // This correction modifies the dynamics to enable a projection of
591  // the model coordinates back onto the constraint manifold whenever
592  // they deviate.
593  // Posa, Kuindersma, Tedrake, 2016. "Optimization and stabilization
594  // of trajectories for constrained dynamical systems"
595  // Note: Only supported for the Hermite-Simpson transcription scheme.
596  if (diffuses.size() != 0) {
597  SimTK::Vector gamma((int)diffuses.size(), diffuses.data());
598  const auto& matter = modelBase.getMatterSubsystem();
599  matter.multiplyByGTranspose(simTKStateBase, gamma, this->qdotCorr);
600  // It doesn't matter what state we use for U since it's U is the
601  // same in both states.
602  this->qdot = simTKStateDisabledConstraints.getU() + this->qdotCorr;
603  } else {
604  this->qdot = simTKStateDisabledConstraints.getU();
605  }
606 
607  // Copy state derivative values to output struct. We cannot simply
608  // use getYDot() because we may have applied a velocity correction to
609  // qdot.
610  const int nq = this->qdot.size();
611  const auto& udot = simTKStateDisabledConstraints.getUDot();
612  const auto& zdot = simTKStateDisabledConstraints.getZDot();
613  const int nu = udot.size();
614  const int nz = zdot.size();
615  std::copy_n(
616  this->qdot.getContiguousScalarData(), nq, out.dynamics.data());
617  std::copy_n(
618  udot.getContiguousScalarData(), nu, out.dynamics.data() + nq);
619  std::copy_n(zdot.getContiguousScalarData(), nz,
620  out.dynamics.data() + nq + nu);
621 
622  // Path constraint errors.
623  this->calcPathConstraintErrors(simTKStateDisabledConstraints, out);
624  }
625 };

627 template <typename T>
628 class MocoTropterSolver::ImplicitTropterProblem
629  : public MocoTropterSolver::TropterProblemBase<T> {
630 public:
631  ImplicitTropterProblem(const MocoTropterSolver& solver)
632  : TropterProblemBase<T>(solver, true) {
633  OPENSIM_THROW_IF(this->m_numKinematicConstraintEquations, Exception,
634  "Cannot use implicit dynamics mode with kinematic "
635  "constraints.");
636 
637  auto& simTKStateDisabledConstraints = this->m_stateDisabledConstraints;
638  if (!this->m_mocoProbRep.isPrescribedKinematics()) {
639  const auto& accel = this->m_mocoProbRep.getAccelerationMotion();
640  accel.setEnabled(simTKStateDisabledConstraints, true);
641  }
642 
643  // Add adjuncts for udot, which we call "w".
644  int NU = simTKStateDisabledConstraints.getNU();
645  for (int iudot = 0; iudot < NU; ++iudot) {
646  auto name = this->m_svNamesInSysOrder[iudot];
647  auto leafpos = name.find("value");
648  OPENSIM_THROW_IF(
649  leafpos == std::string::npos, Exception, "Internal error.");
650  name.replace(leafpos, name.size(), "accel");
651  this->add_adjunct(name,
652  convertBounds(
653  solver.get_implicit_multibody_acceleration_bounds()));
654  this->add_path_constraint(name.substr(0, leafpos) + "residual", 0);
655  }
656  }
657  void calc_differential_algebraic_equations(const tropter::Input<T>& in,
658  tropter::Output<T> out) const override {
659 
660  const auto& states = in.states;
661  const auto& adjuncts = in.adjuncts;
662 
663  const auto& modelDisabledConstraints = this->m_modelDisabledConstraints;
664  auto& simTKStateDisabledConstraints = this->m_stateDisabledConstraints;
665 
666  const int numEmptySlots =
667  simTKStateDisabledConstraints.getNY() - (int)states.size();
668  const auto NQ = simTKStateDisabledConstraints.getNQ() - numEmptySlots;
669  const auto NU = simTKStateDisabledConstraints.getNU();
670  const auto NZ = simTKStateDisabledConstraints.getNZ();
671 
672  const auto& u = states.segment(NQ, NU);
673  const auto& w =
674  adjuncts.segment(this->m_numKinematicConstraintEquations, NU);
675 
676  // Kinematic differential equations
677  // --------------------------------
678  // qdot = u
679  // TODO does not work for quaternions!
680  out.dynamics.head(NQ) = u;
681 
682  // Multibody dynamics: differential equations
683  // ------------------------------------------
684  // udot = w
685  out.dynamics.segment(NQ, NU) = w;
686 
687  // Multibody dynamics: "F - ma = 0"
688  // --------------------------------
689  this->setSimTKState(in);
690 
691  // TODO: Update to support kinematic constraints, using
692  // this->calcKinematicConstraintForces()
693  this->calcPathConstraintErrors(simTKStateDisabledConstraints, out);
694 
695  if (NZ || out.path.size()) {
696  modelDisabledConstraints.realizeAcceleration(
697  simTKStateDisabledConstraints);
698  }
699 
700  if (NZ) {
701  const auto& zdot = simTKStateDisabledConstraints.getZDot();
702  std::copy_n(zdot.getContiguousScalarData(), zdot.size(),
703  out.dynamics.data() + NQ + NU);
704  }
705 
706  if (out.path.size() != 0) {
707  const auto& matter =
708  this->m_modelDisabledConstraints.getMatterSubsystem();
709  matter.findMotionForces(simTKStateDisabledConstraints, m_residual);
710 
711  double* residualBegin = out.path.data() +
712  this->m_numKinematicConstraintEquations +
713  this->m_numPathConstraintEquations;
714  std::copy_n(m_residual.getContiguousScalarData(), m_residual.size(),
715  residualBegin);
716  }
717  }
718 
719 private:
720  mutable SimTK::Vector m_residual;
721 };
